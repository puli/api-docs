 12: namespace Puli\Repository\Api;

 14: use InvalidArgumentException;

 16: /**
 17:  * Stores {@link Resource} objects.
 18:  *
 19:  * A resource repository is similar to a filesystem. It stores {@link Resource}
 20:  * objects, each of which has a path in the repository:
 21:  *
 22:  * ```php
 23:  * $resource = $repo->get('/css/style.css');
 24:  * ```
 25:  *
 26:  * Resources may have child resources. These can be accessed with
 27:  * {@link listChildren()}:
 28:  *
 29:  * ```php
 30:  * $resource = $repo->get('/css');
 31:  *
 32:  * foreach ($resource->listChildren() as $name => $resource) {
 33:  *     // ...
 34:  * }
 35:  * ```
 36:  *
 37:  * @since  1.0
 38:  * @author Bernhard Schussek <bschussek@gmail.com>
 39:  */
 40: interface ResourceRepository
 41: {
 42:     /**
 43:      * Returns the resource at the given path.
 44:      *
 45:      * @param string $path The path to the resource. Must start with "/". "."
 46:      *                     and ".." segments in the path are supported.
 47:      *
 48:      * @return Resource The resource at this path.
 49:      *
 50:      * @throws ResourceNotFoundException If the resource cannot be found.
 51:      * @throws InvalidArgumentException If the path is invalid. The path must be
 52:      *                                  a non-empty string starting with "/".
 53:      */
 54:     public function get($path);
 55: 
 56:     /**
 57:      * Returns the resources matching a query.
 58:      *
 59:      * @param string $query    A resource query.
 60:      * @param string $language The language of the query. All implementations
 61:      *                         must support the language "glob".
 62:      *
 63:      * @return ResourceCollection The resources matching the query.
 64:      *
 65:      * @throws InvalidArgumentException If the query is invalid.
 66:      * @throws UnsupportedLanguageException If the language is not supported.
 67:      */
 68:     public function find($query, $language = 'glob');
 69: 
 70:     /**
 71:      * Returns whether any resources match a query.
 72:      *
 73:      * @param string $query    A resource query.
 74:      * @param string $language The language of the query. All implementations
 75:      *                         must support the language "glob".
 76:      *
 77:      * @return bool Returns `true` if any resources exist that match the query.
 78:      *
 79:      * @throws InvalidArgumentException If the query is invalid.
 80:      * @throws UnsupportedLanguageException If the language is not supported.
 81:      */
 82:     public function contains($query, $language = 'glob');
 83: 
 84:     /**
 85:      * Returns whether a resource has child resources.
 86:      *
 87:      * @param string $path The path to the resource. Must start with "/".
 88:      *                     "." and ".." segments in the path are supported.
 89:      *
 90:      * @return bool Returns `true` if the resource has child resources.
 91:      *
 92:      * @throws ResourceNotFoundException If the resource cannot be found.
 93:      * @throws InvalidArgumentException If the path is invalid. The path must be
 94:      *                                  a non-empty string starting with "/".
 95:      */
 96:     public function hasChildren($path);
 97: 
 98:     /**
 99:      * Lists the child resources of a resource.
100:      *
101:      * @param string $path The path to the resource. Must start with "/".
102:      *                     "." and ".." segments in the path are supported.
103:      *
104:      * @return ResourceCollection The child resources of the resource.
105:      *
106:      * @throws ResourceNotFoundException If the resource cannot be found.
107:      * @throws InvalidArgumentException If the path is invalid. The path must be
108:      *                                  a non-empty string starting with "/".
109:      */
110:     public function listChildren($path);
111: }
