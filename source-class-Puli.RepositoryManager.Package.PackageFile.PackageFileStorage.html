 24: /**
 25:  * Loads and saves package files.
 26:  *
 27:  * This class adds a layer on top of {@link PackageFileReader} and
 28:  * {@link PackageFileWriter}. Any logic that is related to the loading and
 29:  * saving of package configuration, but not directly related to the
 30:  * reading/writing of a specific file format, is executed by this class.
 31:  *
 32:  * The events {@link ManagerEvents::LOAD_PACKAGE_FILE} and
 33:  * {@link ManagerEvents::SAVE_PACKAGE_FILE} are dispatched when a package file
 34:  * is loaded/saved.
 35:  *
 36:  * @since  1.0
 37:  * @author Bernhard Schussek <bschussek@gmail.com>
 38:  */
 39: class PackageFileStorage
 40: {
 41:     /**
 42:      * @var PackageFileReader
 43:      */
 44:     private $reader;
 45: 
 46:     /**
 47:      * @var PackageFileWriter
 48:      */
 49:     private $writer;
 50: 
 51:     /**
 52:      * @var EventDispatcherInterface
 53:      */
 54:     private $dispatcher;
 55: 
 56:     /**
 57:      * Creates a new storage.
 58:      *
 59:      * @param PackageFileReader $reader     The package file reader.
 60:      * @param PackageFileWriter $writer     The package file writer.
 61:      * @param EventDispatcherInterface   $dispatcher The event dispatcher to use.
 62:      */
 63:     public function __construct(PackageFileReader $reader, PackageFileWriter $writer, EventDispatcherInterface $dispatcher)
 64:     {
 65:         $this->reader = $reader;
 66:         $this->writer = $writer;
 67:         $this->dispatcher = $dispatcher;
 68:     }
 69: 
 70:     /**
 71:      * Loads a package file from a file path.
 72:      *
 73:      * If the file does not exist, an empty configuration is returned.
 74:      *
 75:      * The event {@link ManagerEvents::LOAD_PACKAGE_FILE} is dispatched after
 76:      * loading the file. You can attach listeners to this event to modify the
 77:      * loaded file.
 78:      *
 79:      * Loaded package files must have a package name set. If none is set, an
 80:      * exception is thrown.
 81:      *
 82:      * @param string $path The path to the package file.
 83:      *
 84:      * @return PackageFile The loaded package file.
 85:      *
 86:      * @throws InvalidConfigException If the file contains invalid configuration.
 87:      */
 88:     public function loadPackageFile($path)
 89:     {
 90:         try {
 91:             // Don't use file_exists() to decouple from the file system
 92:             $packageFile = $this->reader->readPackageFile($path);
 93:         } catch (FileNotFoundException $e) {
 94:             $packageFile = new PackageFile(null, $path);
 95:         }
 96: 
 97:         if ($this->dispatcher->hasListeners(ManagerEvents::LOAD_PACKAGE_FILE)) {
 98:             $event = new PackageFileEvent($packageFile);
 99:             $this->dispatcher->dispatch(ManagerEvents::LOAD_PACKAGE_FILE, $event);
100:         }
101: 
102:         return $packageFile;
103:     }
104: 
105:     /**
106:      * Saves a package file.
107:      *
108:      * The package file is saved to the same path that it was read from.
109:      *
110:      * The event {@link ManagerEvents::SAVE_PACKAGE_FILE} is dispatched just
111:      * before saving the file. You can attach listeners to this event to modify
112:      * the saved file.
113:      *
114:      * @param PackageFile $packageFile The package file to save.
115:      *
116:      * @throws IOException If the file cannot be written.
117:      */
118:     public function savePackageFile(PackageFile $packageFile)
119:     {
120:         if ($this->dispatcher->hasListeners(ManagerEvents::SAVE_PACKAGE_FILE)) {
121:             $event = new PackageFileEvent($packageFile);
122:             $this->dispatcher->dispatch(ManagerEvents::SAVE_PACKAGE_FILE, $event);
123:         }
124: 
125:         $this->writer->writePackageFile($packageFile, $packageFile->getPath());
126:     }
127: 
128:     /**
129:      * Loads a root package file from a file path.
130:      *
131:      * If the file does not exist, an empty configuration is returned.
132:      *
133:      * The event {@link ManagerEvents::LOAD_PACKAGE_FILE} is dispatched after
134:      * loading the file. You can attach listeners to this event to modify the
135:      * loaded file.
136:      *
137:      * @param string $path       The path to the package configuration file.
138:      * @param Config $baseConfig The configuration that the package will inherit
139:      *                           its configuration values from.
140:      *
141:      * @return RootPackageFile The loaded package file.
142:      *
143:      * @throws InvalidConfigException If the file contains invalid configuration.
144:      */
145:     public function loadRootPackageFile($path, Config $baseConfig)
146:     {
147:         try {

155:             $event = new PackageFileEvent($packageFile);

157:         }

169:      * the saved file.
170:      *
171:      * @param RootPackageFile $packageFile The package file to save.

174:      */
175:     public function saveRootPackageFile(RootPackageFile $packageFile)
