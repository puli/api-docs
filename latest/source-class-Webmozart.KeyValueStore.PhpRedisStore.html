 38:     /**
 39:      * Creates a store backed by a PhpRedis client.
 40:      *
 41:      * If no client is passed, a new one is created using the default server
 42:      * "127.0.0.1" and the default port 6379.
 43:      *
 44:      * @param Redis $client The client used to connect to Redis.
 45:      */
 46:     public function __construct(Redis $client = null)
 47:     {
 48:         if (null === $client) {
 49:             $client = new Redis();
 50:             $client->connect('127.0.0.1', 6379);
 51:         }
 52: 
 53:         $this->client = $client;
 54:     }
 55: 
 56:     /**
 57:      * {@inheritdoc}
 58:      */
 59:     public function set($key, $value)
 60:     {
 61:         KeyUtil::validate($key);
 62: 
 63:         $serialized = Serializer::serialize($value);
 64: 
 65:         try {
 66:             $this->client->set($key, $serialized);
 67:         } catch (Exception $e) {
 68:             throw WriteException::forException($e);
 69:         }
 70:     }
 71: 
 72:     /**
 73:      * {@inheritdoc}
 74:      */
 75:     public function get($key, $default = null)
 76:     {
 77:         KeyUtil::validate($key);
 78: 
 79:         try {
 80:             if (!$this->client->exists($key)) {
 81:                 return $default;
 82:             }
 83: 
 84:             $serialized = $this->client->get($key);
 85:         } catch (Exception $e) {
 86:             throw ReadException::forException($e);
 87:         }
 88:

 90:     }
