 19: use Puli\WebResourcePlugin\Api\WebResourcePlugin;

 22: /**
 23:  * A resource URL generator that uses a {@link ResourceDiscovery} as backend.
 24:  *
 25:  * @since  1.0
 26:  * @author Bernhard Schussek <bschussek@gmail.com>
 27:  */
 28: class DiscoveryUrlGenerator implements ResourceUrlGenerator
 29: {
 30:     /**
 31:      * @var ResourceDiscovery
 32:      */
 33:     private $discovery;
 34: 
 35:     /**
 36:      * @var InstallTargetCollection
 37:      */
 38:     private $targets;
 39: 
 40:     /**
 41:      * Creates the URL generator.
 42:      *
 43:      * @param ResourceDiscovery       $discovery The resource discovery.
 44:      * @param InstallTargetCollection $targets   The available targets.
 45:      */
 46:     public function __construct(ResourceDiscovery $discovery, InstallTargetCollection $targets)
 47:     {
 48:         $this->discovery = $discovery;
 49:         $this->targets = $targets;
 50:     }
 51: 
 52:     /**
 53:      * {@inheritdoc}
 54:      */
 55:     public function generateUrl($repositoryPath, $currentUrl = null)
 56:     {
 57:         $bindings = $this->discovery->findByPath($repositoryPath, WebResourcePlugin::BINDING_TYPE);
 58:         $count = count($bindings);
 59: 
 60:         if (0 === $count) {
 61:             throw new CannotGenerateUrlException(sprintf(
 62:                 'No web path mapping exists for path "%s".',
 63:                 $repositoryPath
 64:             ));
 65:         }
 66: 
 67:         // We can't prevent a resource to be mapped to more than one web path
 68:         // For now, we'll just take the first one and make the user responsible
 69:         // for preventing duplicates
 70:         $url = $this->generateUrlForBinding(reset($bindings), $repositoryPath);
 71: 
 72:         if ($currentUrl) {
 73:             // TODO use Url::makeRelative() once it exists
 74:         }
 75: 
 76:         return $url;
 77:     }
 78: 
 79:     private function generateUrlForBinding(ResourceBinding $binding, $repositoryPath)
 80:     {
 81:         $bindingPath = Glob::getStaticPrefix($binding->getQuery());
 82:         $webBasePath = trim($binding->getParameterValue(WebResourcePlugin::PATH_PARAMETER), '/');
 83:         $webPath = substr_replace($repositoryPath, $webBasePath, 0, strlen($bindingPath));
 84: 
 85:         $targetName = $binding->getParameterValue(WebResourcePlugin::TARGET_PARAMETER);
 86: 
 87:         if (!$this->targets->contains($targetName)) {
 88:             throw new CannotGenerateUrlException(sprintf(
 89:                 'The target "%s" mapped for path "%s" does not exist.',
 90:                 $targetName,
 91:                 $repositoryPath
 92:             ));
 93:         }
 94: 
 95:         $target = $this->targets->get($targetName);
 96: 
 97:         return sprintf($target->getUrlFormat(), ltrim($webPath, '/'));
 98:     }
 99: }
