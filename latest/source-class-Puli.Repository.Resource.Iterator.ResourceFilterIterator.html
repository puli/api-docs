 14: use Assert\Assertion;

 18: /**
 19:  * Iterates over a {@link ResourceIterator} and filters out individual entries.
 20:  *
 21:  * You can use the iterator to filter files with a specific extension:
 22:  *
 23:  * ```php
 24:  * $iterator = new ResourceFilterIterator(
 25:  *     new RecursiveResourceIteratorIterator(
 26:  *         new ResourceCollectionIterator($collection),
 27:  *     ),
 28:  *     '.css',
 29:  *     ResourceFilterIterator::MATCH_SUFFIX
 30:  * );
 31:  *
 32:  * foreach ($iterator as $path => $resource) {
 33:  *     // ...
 34:  * }
 35:  * ```
 36:  *
 37:  * See {@link __construct} for more information on the filter options.
 38:  *
 39:  * @since  1.0
 40:  * @author Bernhard Schussek <bschussek@gmail.com>
 41:  */
 42: class ResourceFilterIterator extends FilterIterator implements ResourceIterator
 43: {
 44:     /**
 45:      * Matches the pattern against the resource path.
 46:      */
 47:     const FILTER_BY_PATH = 1;
 48: 
 49:     /**
 50:      * Matches the pattern against the resource name.
 51:      */
 52:     const FILTER_BY_NAME = 2;
 53: 
 54:     /**
 55:      * Includes resources if the pattern is a prefix of the matched text.
 56:      */
 57:     const MATCH_PREFIX = 32;
 58: 
 59:     /**
 60:      * Includes resources if the pattern is a suffix of the matched text.
 61:      */
 62:     const MATCH_SUFFIX = 64;
 63: 
 64:     /**
 65:      * Includes resources if the matched text satisfies the pattern as regular
 66:      * expression.
 67:      */
 68:     const MATCH_REGEX = 128;
 69: 
 70:     /**
 71:      * @var string
 72:      */
 73:     private $pattern;
 74: 
 75:     /**
 76:      * @var int
 77:      */
 78:     private $patternLength;
 79: 
 80:     /**
 81:      * @var int
 82:      */
 83:     private $mode;
 84: 
 85:     /**
 86:      * Creates a new iterator.
 87:      *
 88:      * The following constants can be used to configure what to filter by:
 89:      *
 90:      *  * {@link FILTER_BY_PATH}: The pattern is matched against the paths;
 91:      *  * {@link FILTER_BY_NAME}: The pattern is matched against the names.
 92:      *
 93:      * The following constants can be used to configure how to match the
 94:      * selected text:
 95:      *
 96:      *  * {@link MATCH_PREFIX}: Tests whether the pattern is a prefix of the
 97:      *                          matched text;
 98:      *  * {@link MATCH_SUFFIX}: Tests whether the pattern is a suffix of the
 99:      *                          matched text;
100:      *  * {@link MATCH_REGEX}: Treats the pattern as regular expression.
101:      *
102:      * By default, the mode `FILTER_BY_PATH | MATCH_REGEX` is used.
103:      *
104:      * @param ResourceIterator $iterator The filtered iterator.
105:      * @param string           $pattern  The pattern to match.
106:      * @param int|null         $mode     A bitwise combination of the mode
107:      *                                   constants.
108:      */
109:     public function __construct(ResourceIterator $iterator, $pattern, $mode = null)
110:     {
111:         Assertion::string($pattern, 'The pattern must be a string. Got: %2$s');
112:         Assertion::notEmpty($pattern, 'The pattern must not be empty');
113: 
114:         parent::__construct($iterator);
115: 
116:         if (!($mode & (self::FILTER_BY_PATH | self::FILTER_BY_NAME))) {
117:             $mode |= self::FILTER_BY_PATH;
118:         }
119: 
120:         if (!($mode & (self::MATCH_PREFIX | self::MATCH_SUFFIX | self::MATCH_REGEX))) {
121:             $mode |= self::MATCH_REGEX;
122:         }
123: 
124:         $this->pattern = $pattern;
125:         $this->patternLength = strlen($pattern);
126:         $this->mode = $mode;
127:     }
128: 
129:     /**
130:      * Returns whether the current element should be accepted.
131:      *
132:      * @return bool Returns `false` if the current element should be filtered out.
133:      */
134:     public function accept()
135:     {
136:         if ($this->mode & self::FILTER_BY_PATH) {
137:             $value = $this->getCurrentResource()->getPath();
138:         } else {
139:             $value = $this->getCurrentResource()->getName();
140:         }
141: 
142:         if ($this->mode & self::MATCH_PREFIX) {
143:             return 0 === strpos($value, $this->pattern);
144:         } elseif ($this->mode & self::MATCH_SUFFIX) {
145:             return $this->pattern === substr($value, -$this->patternLength);
146:         } else {
147:             return preg_match($this->pattern, $value);
148:         }
149:     }
150: 
151:     /**
152:      * {@inheritdoc}
153:      */
154:     public function getCurrentResource()
155:     {
156:         return $this->getInnerIterator()->getCurrentResource();
157:     }
158: }
