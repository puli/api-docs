230:     /**
231:      * Adds a plugin class.
232:      *
233:      * The plugin class must be a fully-qualified class name that implements
234:      * {@link ManagerPlugin}. If the class is not found or does not implement
235:      * that interface, an exception is thrown.
236:      *
237:      * The plugin class must not have required parameters in its constructor
238:      * so that it can be successfully instantiate. If the constructor has
239:      * required parameters, an exception is thrown.
240:      *
241:      * Leading backslashes are removed from the fully-qualified class name.
242:      *
243:      * @param string $pluginClass The fully qualified plugin class name.
244:      *
245:      * @throws InvalidConfigException If the class is not found, is not a class,
246:      *                                does not implement {@link ManagerPlugin}
247:      *                                or has required constructor parameters.
248:      */
249:     public function addPluginClass($pluginClass)
250:     {
251:         try {
252:             $reflClass = new ReflectionClass($pluginClass);
253:         } catch (ReflectionException $e) {
254:             throw new InvalidConfigException(sprintf(
255:                 'The plugin class %s does not exist.',
256:                 $pluginClass
257:             ), 0, $e);
258:         }
259: 
260:         if ($reflClass->isInterface()) {
261:             throw new InvalidConfigException(sprintf(
262:                 'The plugin class %s should be a class, but is an interface.',
263:                 $pluginClass
264:             ));
265:         }
266: 
267:         if (version_compare(PHP_VERSION, '5.4.0', '>=') && $reflClass->isTrait()) {
268:             throw new InvalidConfigException(sprintf(
269:                 'The plugin class %s should be a class, but is a trait.',
270:                 $pluginClass
271:             ));
272:         }
273: 
274:         if (!$reflClass->implementsInterface('\Puli\RepositoryManager\Plugin\ManagerPlugin')) {
275:             throw new InvalidConfigException(sprintf(
276:                 'The plugin class %s must implement ManagerPlugin.',
277:                 $pluginClass
278:             ));
279:         }
280: 
281:         $constructor = $reflClass->getConstructor();
282: 
283:         if (null !== $constructor && $constructor->getNumberOfRequiredParameters() > 0) {
284:             throw new InvalidConfigException(sprintf(
285:                 'The constructor of the plugin class %s must not have required '.
286:                 'parameters.',
287:                 $pluginClass
288:             ));
289:         }
290: 
291:         $this->pluginClasses[ltrim($pluginClass, '\\')] = true;
292:     }
