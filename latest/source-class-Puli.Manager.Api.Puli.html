 57: /**
 58:  * The Puli service locator.
 59:  *
 60:  * Use this class to access the managers provided by this package:
 61:  *
 62:  * ```php
 63:  * $puli = new Puli(getcwd());
 64:  * $puli->start();
 65:  *
 66:  * $packageManager = $puli->getPackageManager();
 67:  * ```
 68:  *
 69:  * The `Puli` class either operates in the global or a project environment:
 70:  *
 71:  *  * The "global environment" is not tied to a specific root package. A global
 72:  *    environment only loads the settings of the "config.json" file in the home
 73:  *    directory. The `Puli` class operates in the global environment if no
 74:  *    project root directory is passed to the constructor. In the global
 75:  *    environment, only the global config file manager is available.
 76:  *  * The "project environment" is tied to a specific Puli project. You need to
 77:  *    pass the path to the project's root directory to the constructor or to
 78:  *    {@link setRootDirectory()}. The configuration of the "puli.json" file in
 79:  *    the root directory is used to configure the managers.
 80:  *
 81:  * The `Puli` class creates four kinds of managers:
 82:  *
 83:  *  * The "config file manager" allows you to modify entries of the
 84:  *    "config.json" file in the home directory.
 85:  *  * The "package file manager" manages modifications to the "puli.json" file
 86:  *    of a Puli project.
 87:  *  * The "package manager" manages the package repository of a Puli project.
 88:  *  * The "repository manager" manages the resource repository of a Puli
 89:  *    project.
 90:  *  * The "discovery manager" manages the resource discovery of a Puli project.
 91:  *
 92:  * The home directory is read from the environment variable "PULI_HOME".
 93:  * If this variable is not set, the home directory defaults to:
 94:  *
 95:  *  * `$HOME/.puli` on Linux, where `$HOME` is the environment variable
 96:  *    "HOME".
 97:  *  * `$APPDATA/Puli` on Windows, where `$APPDATA` is the environment
 98:  *    variable "APPDATA".
 99:  *
100:  * If none of these variables can be found, an exception is thrown.
101:  *
102:  * A .htaccess file is put into the home directory to protect it from web
103:  * access.
104:  *
105:  * @since  1.0
106:  * @author Bernhard Schussek <bschussek@gmail.com>
107:  */

611:     private function createGlobalEnvironment()
612:     {
613:         $homeDir = self::parseHomeDirectory();
614: 
615:         if (null !== $homeDir) {
616:             Assert::fileExists($homeDir, 'Could not load Puli environment: The home directory %s does not exist.');
617:             Assert::directory($homeDir, 'Could not load Puli environment: The home directory %s is a file. Expected a directory.');
618: 
619:             // Create a storage without the factory manager
620:             $configStorage = new ConfigFileStorage($this->getConfigFileReader(), $this->getConfigFileWriter());
621:             $configPath = Path::canonicalize($homeDir).'/config.json';
622:             $configFile = $configStorage->loadConfigFile($configPath, new DefaultConfig());
623:             $baseConfig = $configFile->getConfig();
624:         } else {
625:             $configFile = null;
626:             $baseConfig = new DefaultConfig();
627:         }
628: 
629:         $config = new EnvConfig($baseConfig);
630: 
631:         return new GlobalEnvironment($homeDir, $config, $configFile, $this->dispatcher);
632:     }

634:     /**
635:      * Creates the environment of a Puli project.
636:      *
637:      * The home directory is read from the environment variable "PULI_HOME".
638:      * If this variable is not set, the home directory defaults to:
639:      *
640:      *  * `$HOME/.puli` on Linux, where `$HOME` is the environment variable
641:      *    "HOME".
642:      *  * `$APPDATA/Puli` on Windows, where `$APPDATA` is the environment
643:      *    variable "APPDATA".
644:      *
645:      * If none of these variables can be found, an exception is thrown.
646:      *
647:      * A .htaccess file is put into the home directory to protect it from web
648:      * access.
649:      *
650:      * @param string $rootDir The path to the project.
651:      *
652:      * @return ProjectEnvironment The project environment.
653:      */
654:     private function createProjectEnvironment($rootDir)
655:     {
656:         Assert::fileExists($rootDir, 'Could not load Puli environment: The root %s does not exist.');
657:         Assert::directory($rootDir, 'Could not load Puli environment: The root %s is a file. Expected a directory.');
658: 
659:         $homeDir = self::parseHomeDirectory();
660: 
661:         if (null !== $homeDir) {
662:             Assert::fileExists($homeDir, 'Could not load Puli environment: The home directory %s does not exist.');
663:             Assert::directory($homeDir, 'Could not load Puli environment: The home directory %s is a file. Expected a directory.');
664: 
665:             // Create a storage without the factory manager
666:             $configStorage = new ConfigFileStorage($this->getConfigFileReader(), $this->getConfigFileWriter());
667:             $configPath = Path::canonicalize($homeDir).'/config.json';
668:             $configFile = $configStorage->loadConfigFile($configPath, new DefaultConfig());
669:             $baseConfig = $configFile->getConfig();
670:         } else {
671:             $configFile = null;
672:             $baseConfig = new DefaultConfig();
673:         }
674: 
675:         // Create a storage without the factory manager
676:         $packageFileStorage = new PackageFileStorage($this->getPackageFileReader(), $this->getPackageFileWriter());
677:         $rootDir = Path::canonicalize($rootDir);
678:         $rootFilePath = $this->rootDir.'/puli.json';
679:         $rootPackageFile = $packageFileStorage->loadRootPackageFile($rootFilePath, $baseConfig);
680:         $config = new EnvConfig($rootPackageFile->getConfig());
681: 
682:         return new ProjectEnvironment($homeDir, $rootDir, $config, $rootPackageFile, $configFile, $this->dispatcher);
683:     }
