 70:     /**
 71:      * Creates the project environment.
 72:      *
 73:      * The passed home directory will be be scanned for a file "config.json".
 74:      * If that file exists, it is loaded into memory. Use {@link getConfig()} to
 75:      * access the project configuration.
 76:      *
 77:      * The passed root directory will be be scanned for a file "puli.json".
 78:      * If that file exists, it is loaded into memory. Use
 79:      * {@link getRootPackageFile()} to access the package file.
 80:      *
 81:      * @param string                   $homeDir            The path to Puli's home directory.
 82:      * @param string                   $rootDir            The path to the project directory.
 83:      * @param ConfigFileStorage        $configFileStorage  The global config file storage.
 84:      * @param PackageFileStorage       $packageFileStorage The package file storage.
 85:      * @param EventDispatcherInterface $dispatcher         The event dispatcher.
 86:      *
 87:      * @throws FileNotFoundException If the home/project directory does not exist.
 88:      * @throws NoDirectoryException If the home/project directory is not a directory.
 89:      */
 90:     public function __construct($homeDir, $rootDir, ConfigFileStorage $configFileStorage, PackageFileStorage $packageFileStorage, EventDispatcherInterface $dispatcher)
 91:     {
 92:         if (!file_exists($rootDir)) {
 93:             throw new FileNotFoundException(sprintf(
 94:                 'Could not load Puli environment: The root %s does not exist.',
 95:                 $rootDir
 96:             ));
 97:         }
 98: 
 99:         if (!is_dir($rootDir)) {
100:             throw new NoDirectoryException(sprintf(
101:                 'Could not load Puli environment: The root %s is a file. '.
102:                 'Expected a directory.',
103:                 $rootDir
104:             ));
105:         }
106: 
107:         parent::__construct($homeDir, $configFileStorage, $dispatcher);
108: 
109:         // Get the global config.json, if any
110:         $configFile = $this->getConfigFile();
111: 
112:         $this->rootDir = Path::canonicalize($rootDir);
113:         $this->rootPackageFile = $packageFileStorage->loadRootPackageFile(
114:             $this->rootDir.'/puli.json',
115:             $configFile ? $configFile->getConfig() : new DefaultConfig()
116:         );
117: 
118:         // Override the configuration with the one from the root package
119:         $this->setConfig(new EnvConfig($this->rootPackageFile->getConfig()));
120:     }

170:     private function loadFactory()
171:     {
172:         $factoryFile = $this->getConfig()->get(Config::FACTORY_FILE);
173:         $factoryClass = $this->getConfig()->get(Config::FACTORY_CLASS);
174: 
175:         Assert::string($factoryFile, 'The "'.Config::FACTORY_FILE.'" config key must contain a string. Got: %s');
176:         Assert::notEmpty($factoryFile, 'The "'.Config::FACTORY_FILE.'" config key must not be empty.');
177:         Assert::string($factoryClass, 'The "'.Config::FACTORY_CLASS.'" config key must contain a string. Got: %s');
178:         Assert::notEmpty($factoryClass, 'The "'.Config::FACTORY_CLASS.'" config key must not be empty.');
179: 
180:         if (!class_exists($factoryClass, false)) {
181:             $autoGenerate = $this->getConfig()->get(Config::FACTORY_AUTO_GENERATE);
182:             $absFactoryFile = Path::makeAbsolute($factoryFile, $this->rootDir);
183:             $lastConfigChange = 0;
184:             $lastFactoryUpdate = 0;
185: 
186:             // Regenerate file if the configuration has changed and
187:             // auto-generation is enabled
188:             if (file_exists($absFactoryFile) && $autoGenerate) {
189:                 clearstatcache(true, $this->rootPackageFile->getPath());
190:                 $lastConfigChange = filemtime($this->rootPackageFile->getPath());
191: 
192:                 if ($this->getConfigFile() && file_exists($this->getConfigFile()->getPath())) {
193:                     clearstatcache(true, $this->getConfigFile()->getPath());
194:                     $lastConfigChange = max(filemtime($this->getConfigFile()->getPath()), $lastConfigChange);
195:                 }
196: 
197:                 clearstatcache(true, $absFactoryFile);
198:                 $lastFactoryUpdate = filemtime($absFactoryFile);
199:             }
200: 
201:             if (!file_exists($absFactoryFile) || $lastConfigChange > $lastFactoryUpdate) {
202:                 $generator = new PuliFactoryGenerator();
203: 
204:                 $generator->generateFactory(
205:                     $absFactoryFile,
206:                     $factoryClass,
207:                     $this->rootDir,
208:                     $this->getConfig()
209:                 );
210:             }
211: 
212:             require_once $absFactoryFile;
213:         }
214: 
215:         $this->factory = new $factoryClass;
216:     }
