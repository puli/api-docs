 12: namespace Puli\Manager\Discovery\Type;

 14: use LogicException;
 15: use Puli\Discovery\Api\EditableDiscovery;
 16: use Puli\Manager\Api\Discovery\BindingTypeDescriptor;
 17: use Puli\Manager\Transaction\AtomicOperation;
 18: 
 19: /**
 20:  * Synchronizes a type name with the discovery.
 21:  *
 22:  * The method {@link takeSnapshot()} must be called before executing the
 23:  * operation. This method will record whether the type name is currently
 24:  * enabled (i.e. loaded in the discovery).
 25:  *
 26:  * Once the operation is executed, another snapshot is taken. If the snapshots
 27:  * differ, type is either defined or undefined in the discovery, depending on
 28:  * the outcome.
 29:  *
 30:  * @since  1.0
 31:  * @author Bernhard Schussek <bschussek@gmail.com>
 32:  */
 33: class SyncTypeName implements AtomicOperation
 34: {
 35:     /**
 36:      * @var string
 37:      */
 38:     private $typeName;
 39: 
 40:     /**
 41:      * @var EditableDiscovery
 42:      */
 43:     private $discovery;
 44: 
 45:     /**
 46:      * @var BindingTypeDescriptorCollection
 47:      */
 48:     private $typeDescriptors;
 49: 
 50:     /**
 51:      * @var BindingTypeDescriptor|null
 52:      */
 53:     private $enabledTypeBefore;
 54: 
 55:     /**
 56:      * @var BindingTypeDescriptor|null
 57:      */
 58:     private $enabledTypeAfter;
 59: 
 60:     /**
 61:      * @var bool
 62:      */
 63:     private $snapshotTaken = false;
 64: 
 65:     public function __construct($typeName, EditableDiscovery $discovery, BindingTypeDescriptorCollection $typeDescriptors)
 66:     {
 67:         $this->typeName = $typeName;
 68:         $this->discovery = $discovery;
 69:         $this->typeDescriptors = $typeDescriptors;
 70:     }
 71: 
 72:     /**
 73:      * Records whether the type name is currently enabled.
 74:      */
 75:     public function takeSnapshot()
 76:     {
 77:         $this->enabledTypeBefore = $this->typeDescriptors->getEnabled($this->typeName);
 78:         $this->snapshotTaken = true;
 79:     }
 80: 
 81:     /**
 82:      * {@inheritdoc}
 83:      */
 84:     public function execute()
 85:     {
 86:         if (!$this->snapshotTaken) {
 87:             throw new LogicException('takeSnapshot() was not called');
 88:         }
 89: 
 90:         // Remember for rollback()
 91:         $this->enabledTypeAfter = $this->typeDescriptors->getEnabled($this->typeName);
 92: 
 93:         $this->syncTypeName($this->enabledTypeBefore, $this->enabledTypeAfter);
 94:     }
 95: 
 96:     /**
 97:      * {@inheritdoc}
 98:      */
 99:     public function rollback()
100:     {
101:         $this->syncTypeName($this->enabledTypeAfter, $this->enabledTypeBefore);
102:     }
103: 
104:     private function syncTypeName(BindingTypeDescriptor $enabledTypeBefore = null, BindingTypeDescriptor $enabledTypeAfter = null)
105:     {
106:         if ($enabledTypeBefore && !$enabledTypeAfter) {
107:             $this->discovery->undefineType($this->typeName);
108:         } elseif (!$enabledTypeBefore && $enabledTypeAfter) {
109:             $this->discovery->defineType($enabledTypeAfter->toBindingType());
110:         } elseif ($enabledTypeBefore !== $enabledTypeAfter) {
111:             $this->discovery->undefineType($this->typeName);
112:             $this->discovery->defineType($enabledTypeAfter->toBindingType());
113:         }
114:     }
115: }
